#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>  // para usar std::setprecision

using namespace std;

int main() {
    int N;
    cout << "Digite a quantidade de numeros: ";
    cin >> N;

    vector<int> numeros(N);

    cout << "Digite os " << N << " numeros:\n";
    for (int i = 0; i < N; i++) {
        cin >> numeros[i];
    }

    // Maior e menor
    int maior = *max_element(numeros.begin(), numeros.end());
    int menor = *min_element(numeros.begin(), numeros.end());

    // MÃ©dia
    double soma = 0;
    for (int num : numeros) soma += num;
    double media = soma / N;

    // Ordenar
    sort(numeros.begin(), numeros.end());

    // Resultados
    cout << "\nResultado:\n";
    cout << "Maior numero: " << maior << endl;
    cout << "Menor numero: " << menor << endl;
    cout << "Media: " << fixed << setprecision(2) << media << endl;

    cout << "Numeros em ordem crescente: ";
    for (int num : numeros) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
